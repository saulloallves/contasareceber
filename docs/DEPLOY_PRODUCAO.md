  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  level TEXT NOT NULL, -- info, warning, error
  message TEXT NOT NULL,
  context JSONB,
  created_at TIMESTAMPTZ DEFAULT now()

  p_level TEXT,
  p_message TEXT,
  p_context JSONB DEFAULT '{}'::jsonb
) RETURNS void AS $$

  VALUES (p_level, p_message, p_context);
END;
